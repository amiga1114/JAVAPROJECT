*&---------------------------------------------------------------------*
*&      Form  GET_DATA
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM get_data .
  CLEAR : gt_data, gt_data[].
  SELECT a~matnr b~maktx a~pstat a~mtart a~matkl c~werks d~name1 a~meins
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM mara AS a
    INNER JOIN makt AS b
            ON a~matnr = b~matnr
           AND b~spras EQ 'EN'
    INNER JOIN marc AS c
            ON a~matnr = c~matnr
    INNER JOIN t001w AS d
            ON c~werks = d~werks
    WHERE a~matnr = gv_matnr.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH 'DATA NOT FOUND'.
  ENDIF.
ENDFORM.
*&---------------------------------------------------------------------*

*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM get_init_data .
  IF gt_data[] IS INITIAL.
    CLEAR : gt_data, gt_data[].
    SELECT a~matnr b~maktx a~pstat a~mtart a~matkl c~werks d~name1 a~meins
      INTO CORRESPONDING FIELDS OF TABLE gt_data
      FROM mara AS a
      INNER JOIN makt AS b
              ON a~matnr = b~matnr
             AND b~spras EQ 'EN'
      INNER JOIN marc AS c
              ON a~matnr = c~matnr
      INNER JOIN t001w AS d
              ON c~werks = d~werks.
  ENDIF.

ENDFORM.
