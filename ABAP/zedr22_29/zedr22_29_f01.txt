*  <--  p2        text
*----------------------------------------------------------------------*
FORM save_data .
  CALL METHOD go_grid->check_changed_data.

  IF gt_data[] IS NOT INITIAL.
    MODIFY ztproj22_01 FROM TABLE gt_data. " MODIFY를 사용해 TABLE에 DATA 입력

    IF sy-subrc EQ 0.
      COMMIT WORK AND WAIT. " 다른 일을 하고 있는 경우 기다렸다가 처리함
      MESSAGE s000 WITH 'SAVE SUCCESS'.
      PERFORM get_data.
    ELSE.
      ROLLBACK WORK.
      MESSAGE s000 WITH 'SAVE FAIL' DISPLAY LIKE 'E'.
    ENDIF.
  ENDIF.
ENDFORM.
*&---------------------------------------------------------------------*
*&      Form  SAVE_EDIT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM save_edit .
  DATA : lt_save TYPE TABLE OF ztproj22_02 WITH HEADER LINE.

  DATA : BEGIN OF lt_text OCCURS 0,
           line_text(100),
         END OF lt_text.

  CLEAR : lt_text, lt_text[], lt_save, lt_save[].

  CALL METHOD go_edit2->get_text_as_r3table
    IMPORTING
      table                  = lt_text[]
    EXCEPTIONS
      error_dp               = 1
      error_cntl_call_method = 2
      error_dp_create        = 3
      potential_data_loss    = 4
      OTHERS                 = 5.

  lt_save-seq = gv_seq.
  lt_save-name = gv_name.

  LOOP AT lt_text.
    CONCATENATE lt_save-contents lt_text-line_text cl_abap_char_utilities=>newline
    INTO lt_save-contents.
  ENDLOOP.

  APPEND lt_save.
  CLEAR lt_save.

  MODIFY ztproj22_02 FROM TABLE lt_save.

  IF sy-subrc EQ 0.
    COMMIT WORK AND WAIT.
    MESSAGE s000 WITH 'SAVE SUCCESS'.
  ELSE.
    ROLLBACK WORK.
    MESSAGE s000 WITH 'SAVE FAIL'DISPLAY LIKE 'E'.
  ENDIF.
ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  GET_BUKRS
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->P_0022   text
*----------------------------------------------------------------------*
FORM get_bukrs  USING    p_field..
  DATA : BEGIN OF lt_bukrs OCCURS 0,
           bukrs LIKE t001-bukrs,
           butxt LIKE t001-butxt,
         END OF lt_bukrs.

  CLEAR : lt_bukrs, lt_bukrs[].

  SELECT bukrs butxt
    INTO CORRESPONDING FIELDS OF TABLE lt_bukrs
    FROM t001.

  CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
    EXPORTING
      retfield        = 'BUKRS'
      dynpprog        = sy-repid
      dynpnr          = sy-dynnr
      dynprofield     = p_field
      window_title    = '회사코드목록'
      value_org       = 'S'
    TABLES
      value_tab       = lt_bukrs[]
    EXCEPTIONS
      parameter_error = 1
      no_values_found = 2
      OTHERS          = 3.
ENDFORM.
