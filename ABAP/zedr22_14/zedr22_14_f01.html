FORM get_data .
  CLEAR : gt_data, gt_data[].

  SELECT carrid carrname currcode url
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM scarr
    WHERE carrid IN s_carrid.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH 'DATA NOT FOUND'.
    LEAVE LIST-PROCESSING.
  ENDIF.

ENDFORM.
